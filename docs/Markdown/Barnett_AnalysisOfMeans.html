1-Way Analysis of Variance

Diet Issues:

The underlying linear model.

\[ y_{ij} = \mu + \tau_{Diet, i} + \epsilon_j \]

Test the null hypothesis, \(H_O:\) There is no effect of diet on weight loss (e.g., \(\tau_{Diet-1} = \tau_{Diet-2} = \tau_{Diet-3} = 0.0\)). Is there evidence for one diet producing more weight loss than the others? Create an aov analysis and assign it to the variable fit.diet and examine its contents.

# Define mode
loss$Diet <- as.factor(loss$Diet)

fit.diet <- aov(WeightLoss ~ Diet, data = loss)

#LOOKING FOR AIC??

response <- "WeightLoss"
predictors <- setdiff(names(loss), response)

all_combinations <- unlist(
  lapply(0:length(predictors), function(x) combn(predictors, x, simplify = FALSE)), 
  recursive = FALSE
)

model_results <- lapply(all_combinations, function(predictor_set) {
  predictor_set_escaped <- ifelse(length(predictor_set) > 0, 
                                  paste0("`", predictor_set, "`"), 
                                  "1")
  formula <- as.formula(
    paste0("`", response, "` ~ ", paste(predictor_set_escaped, collapse = " + "))
  )
  
  # Fit the model
  model <- lm(formula, data = loss)
  
  list(
    predictors = predictor_set,
    model = model,
    AIC = AIC(model)
  )
})

results_df <- do.call(rbind, lapply(model_results, function(x) {
  data.frame(
    Predictors = paste(x$predictors, collapse = " + "),
    AIC = x$AIC
  )
}))

min_AIC <- min(results_df$AIC)

results_df <- results_df %>%
  mutate(Delta_AIC = AIC - min_AIC) %>%
  arrange(Delta_AIC)  # Sort by delta AIC

# Print the results with Delta AIC
